[docs]def createFolderStructure(rootDirectory, user):
    """
    Creates Directory structure if it is not already present
    """
    fullPath = rootDirectory + "/"+ user
   
    try:
        os.makedirs(fullPath)
        os.makedirs(fullPath + "/images/")
